# docker安装gitlab
docker run \
--restart=always \
--name hujz.gitlab \
--network hujz \
-v /root/gitlab:/etc/gitlab \
-v /data/gitlab/opt:/var/opt/gitlab \
-v /data/gitlab/backups:/var/opt/gitlab/backups \
-v /etc/localtime:/etc/localtime:ro \
-d beginor/gitlab-ce:10.7.3-ce.0
